# import numpy as np
resumen_mensual = (
    tmx.groupby(pd.Grouper(freq="ME"))
       .agg(
           Ig_sum  = ("Ig", "sum"),          # energía/irradiación acumulada (si aplica)
           To_mean = ("To", "mean"),         # temp. media mensual
           ws_max  = ("ws", "max"),          # racha máxima
           uv_p95  = ("uv", lambda s: s.quantile(0.95))  # percentil 95 de UV
       )
       .rename_axis("Mes")
)
